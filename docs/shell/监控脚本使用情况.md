\#!/bin/bash  
监控磁盘 1个小时监控一次  
dosomething()  
{  
cat \<\<EOF \>a.tmp  
kjkjlkjsfjksjkjss  
sfsfsf  
EOF  
}  
cat a.tmp \|mail -s "zhuti" 123\@qq.com  
while :  
do  
max_num=50  
df_num=\`df -h \|grep /dev/sda \|awk -F'[ %]+' '{print \$5}'\`  
[ \$df_num -ge \$max_num ] && dosomething  
sleep 3600  
done

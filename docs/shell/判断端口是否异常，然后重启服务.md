\#!/bin/bash  
NTOP_PORT=\`netstat -lpnt \| grep 3000 \| awk '{print \$4}' \| awk -F ":"
'{print \$NF}'\`  
if [ \$NTOP_PORT -eq "3000" ];then  
echo "ntopng already running \`date\`"  
else  
/etc/init.d/redis restart  
sleep 5  
/usr/local/ntopng/bin/ntopng /usr/local/ntopng/etc/ntopng.conf &  
fi

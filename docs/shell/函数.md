\#!/bin/bash  
\#定义一个颜色输出字符串函数1  
echo_color(){  
if [ \$1 == "green" ]; then  
echo -e "\\033[32;40m\$2\\033[0m"  
elif [ \$1 == "red" ]; then  
echo -e "\\033[32;40m\$2\\033[0m"  
fi  
}  
echo_color green "yanse"

\#!/bin/bash  
isnum(){  
a=\$1  
expr \$a + 1 \>/dev/null 2\>&1  
[ \$? -ne 0 ] && echo "\$a 不是数字" \|\| echo "\$a 是数字"  
}
